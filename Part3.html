state_url = 'https://www.epa.gov/superfund/national-priorities-list-npl-sites-state'

# get text
def GetHtmlText(url):
    res = req.get(url)
    s = res.raise_for_status()
    res.encoding = res.apparent_encoding
    return res.text

state_dic = GetHtmlText(state_url)

text = GetHtmlText(state_url)

soup = BeautifulSoup(text,'lxml')

# find this specific class contained the information
div = soup.find('div',class_ = 'node node-page clearfix view-mode-full')

id_list = div.find('select',id='list').find_all('option')

state_dic = {}

# then we get the state on this epa website
for i in id_list:
    state = []
    Shortname = i['value'].strip("#")
    for j in allRegion:
        if Shortname in allRegion[j]:
            state.append(j)
            break
    state.append(i.string)
    state_dic[Shortname] = state

sitename=list()

Federal_Facility_Indicator=list()
Partial_Deletion=list()
Construction=list()

# after inspecting element, I find this web page on the Network containing every thing I need
url = 'https://www.epa.gov/sites/production/files/2015-08/nplfin.xml'

text2 = GetHtmlText(url)

soup = BeautifulSoup(text2,'html.parser')

ss = soup.find_all('s')

# regex
findSiteName = re.compile('<d1><!\[CDATA\[(.*?)\]]')

# after looking through the url above, appending each information into the empty list
for s in ss:
    d1 = s.find_all('d1')[0]
    SiteName = re.findall(findSiteName,str(s))[0].strip()
    sitename.append(SiteName)
    score.append(s['o'])
    Site_EPA_ID.append(s["e"])
    shortname = s["a"]
    regionID.append(state_dic[shortname][0])
    city.append(s["l"])
    state.append(state_dic[shortname][1])
    Latitude.append(s["j"])
    Longitude.append(s["k"])
    Listing.append(s["p"])
    Proposed.append(s["f"])
    status.append("NPL Site")
    
    if 'h' in s.attrs:
        Federal_Facility_Indicator.append('yes')
    else:
        Federal_Facility_Indicator.append('no')
    
    if len(s.find_all('d2')) == 1:
        Partial_Deletion.append('yes')
    else:
        Partial_Deletion.append('no')
    if 'c' in s.attrs:
        Construction.append(s['c'])
    else:
        Construction.append('')

# make several lists into dataframe

data = {"sitename":sitename,"score":score,"Site_EPA_ID":Site_EPA_ID,
            "regionID":regionID,"city":city,"state":state,
            "Latitude":Latitude,"Longitude":Longitude,"Listing":Listing,
            "Proposed":Proposed,"status":status,
            "Federal_Facility_Indicator":Federal_Facility_Indicator,
            "Partial_Deletion":Partial_Deletion,"Construction":Construction}

superfund = pd.DataFrame(data)

# split these features into year, month and day

def parse_date(name):
    superfund[name + '_year'] = [i[0:4] for i in superfund[name]]
    superfund[name + '_month'] = [i[4:6] for i in superfund[name]]
    superfund[name + '_day'] = [i[6:8] for i in superfund[name]]
    superfund.drop(name,axis=1,inplace=True)
